front-end bygg (gulp)


gulp.task('iconFont', function () {
    return gulp.src(_.path.svgIcons)
        .pipe(iconfontCss({
            fontName: fontName,
            path: './gulpscripts/assets/_iconsTemplate.less',
            targetPath: '../../Styles/Global/icons-compiled.less',
            fontPath: '/Content/Fonts/Icons/'
        }))
        .pipe(iconfont({
            fontName: fontName,
            path: 'Content/Fonts/Icons/'
        }))
        .pipe(gulp.dest('Content/Fonts/Icons'));
});
